# for song meanings

CREATE TABLE `song_meaning` (
`SongID` INT(25) NOT NULL PRIMARY KEY,
`Meaning` VARCHAR(2048) NOT NULL,
`UserID` INT(25) NOT NULL
);

# for song meanings queue

CREATE TABLE `song_meaning_queue` (
`SubmissionID` INT(25) NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`Processed` BOOLEAN DEFAULT FALSE,
`TimeStamp` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
`SongID` INT(25) NOT NULL,
`Meaning` VARCHAR(2048) NOT NULL,
`UserID` INT(25) NOT NULL
);
